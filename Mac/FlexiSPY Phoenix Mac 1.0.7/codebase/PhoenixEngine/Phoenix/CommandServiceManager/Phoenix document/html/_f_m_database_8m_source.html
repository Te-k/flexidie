00712                 if (rc != SQLITE_LOCKED) {
00713                     NSLog(@"Unexpected result from sqlite3_reset (%d) eu", rc);
00714                 }
00715             }
00716             usleep(20);
00717             
00718             if (busyRetryTimeout && (numberOfRetries++ > busyRetryTimeout)) {
00719                 NSLog(@"%s:%d Database busy (%@)", __FUNCTION__, __LINE__, [self databasePath]);
00720                 NSLog(@"Database busy");
00721                 retry = NO;
00722             }
00723         }
00724         else if (SQLITE_DONE == rc || SQLITE_ROW == rc) {
00725             // all is well, let's return.
00726         }
00727         else if (SQLITE_ERROR == rc) {
00728             NSLog(@"Error calling sqlite3_step (%d: %s) SQLITE_ERROR", rc, sqlite3_errmsg(db));
00729             NSLog(@"DB Query: %@", sql);
00730         }
00731         else if (SQLITE_MISUSE == rc) {
00732             // uh oh.
00733             NSLog(@"Error calling sqlite3_step (%d: %s) SQLITE_MISUSE", rc, sqlite3_errmsg(db));
00734             NSLog(@"DB Query: %@", sql);
00735         }
00736         else {
00737             // wtf?
00738             NSLog(@"Unknown error calling sqlite3_step (%d: %s) eu", rc, sqlite3_errmsg(db));
00739             NSLog(@"DB Query: %@", sql);
00740         }
00741         
00742     } while (retry);
00743     
00744     assert( rc!=SQLITE_ROW );
00745     
00746     
00747     if (shouldCacheStatements && !cachedStmt) {
00748         cachedStmt = [[FMStatement alloc] init];
00749         
00750         [cachedStmt setStatement:pStmt];
00751         
00752         [self setCachedStatement:cachedStmt forQuery:sql];
00753         
00754         [cachedStmt release];
00755     }
00756     
00757     if (cachedStmt) {
00758         cachedStmt.useCount = cachedStmt.useCount + 1;
00759         rc = sqlite3_reset(pStmt);
00760     }
00761     else {
00762         /* Finalize the virtual machine. This releases all memory and other
00763          ** resources allocated by the sqlite3_prepare() call above.
00764          */
00765         rc = sqlite3_finalize(pStmt);
00766     }
00767     
00768     [self setInUse:NO];
00769     
00770     return (rc == SQLITE_OK);
